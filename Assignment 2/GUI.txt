    private void browseButtonActionPerformed(java.awt.event.ActionEvent evt) {
        // TODO add your handling code here:
        try {
            JFileChooser chooser = new JFileChooser();
            chooser.showOpenDialog(null);
            File f = chooser.getSelectedFile();
            String filename = f.getAbsolutePath();
            fileNameTextField.setText(filename);
        } catch (Exception e) {
            jTextArea1.append("===> No file selected" + "\n" + "Please select a file" + "\n\n");
        }
    }

    private void writeFileButtonActionPerformed(java.awt.event.ActionEvent evt) {
        // TODO add your handling code here:
        String s = fileNameTextField.getText();
        File f = new File(s);
        try {
            Scanner scan = new Scanner(f);
            while (scan.hasNextLine()) {
                if (scan.hasNextLine()) {
                    list.add(scan.nextLine());
                } else {
                    scan.next();
                }
            }
            scan.close();

            for (String temp : list) {
                tree.put(temp.substring(0, 7), temp);
            }

            jTextArea1.append("Hight Of the tree: " + tree.height() + "\n");
            jTextArea1.append("Size of Element in tree: " + tree.size() + "\n");
            jTextArea1.append("View of Tree: " + "\n" + tree.toString());

        } catch (FileNotFoundException e) {
            jTextArea1.append("===> FIle not found" + "\n" + "Select new file" + "\n\n");
        }
    }

    private void searchIDButtonActionPerformed(java.awt.event.ActionEvent evt) {
        // TODO add your handling code here:

        String s = patientIDTextField.getText();
        jTextArea2.append("All image scan for patient ID: " + s + "\n");
        List temp = tree.get(s);
        try {
            for (Object val : temp) {
                jTextArea2.append(val + "\n");
            }
        } catch (Exception e) {
            jTextArea2.append("===> No scans for patient ID " + s + " was found" + "\n\n");
        }
    }
